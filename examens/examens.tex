\newcommand{\examnumber}{1}
\newcounter{showall}
\setcounter{showall}{0} % set to 0 to only show one exam

\begin{document}

\selectlanguage{french}

\ifnum \value{showall}<1 {
    %\edef\FileName{exam\examnumber}%

} \else {

        \edef\FileName{exam\c}%
        \IfFileExists{\FileName}{%
        \input{\FileName}%
        }
    }
}
\fi

\end{document}
